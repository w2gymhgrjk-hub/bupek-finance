2. Install Node.js 20

curl -fsSL https://deb.nodesource.com/setup_20.x | sudo bash -
sudo apt-get install -y nodejs
node --version   # should print v20.x.x
npm --version

3. Install & Configure PostgreSQL

sudo apt-get install -y postgresql postgresql-contrib
sudo systemctl enable postgresql
sudo systemctl start postgresql

Create the BUPEK database

sudo -u postgres psql

-- Inside psql:
CREATE USER bupek WITH PASSWORD 'CHOOSE_STRONG_PASSWORD_HERE';
CREATE DATABASE bupek_finance OWNER bupek;
GRANT ALL PRIVILEGES ON DATABASE bupek_finance TO bupek;
\q
⚠ Replace CHOOSE_STRONG_PASSWORD_HERE with a real strong password. Save it — you'll need it for the .env file.

4. Install PM2 (Process Manager)

sudo npm install -g pm2
pm2 --version

PM2 keeps the backend running after you log out of the server, and auto-restarts it if it crashes.

5. Install & Configure Nginx

sudo apt-get install -y nginx
sudo systemctl enable nginx
sudo systemctl start nginx

Create Nginx site configuration

sudo nano /etc/nginx/sites-available/bupek

Paste this into the file:

server {
    listen 80;
    server_name yourdomain.co.tz;   # ← replace with your domain

    # Frontend (Next.js)
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Real-IP $remote_addr;
    }

    # Backend API
    location /api/ {
        proxy_pass http://localhost:5000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        client_max_body_size 10M;
    }

    # File uploads
    location /uploads/ {
        alias /opt/bupek-finance/backend/uploads/;
        expires 30d;
    }
}
# Enable the site
sudo ln -s /etc/nginx/sites-available/bupek /etc/nginx/sites-enabled/
sudo nginx -t          # test config — should say "test is successful"
sudo systemctl reload nginx

6. Upload Application Code

# On your local computer — upload the code to the server
scp -r /path/to/bupek-finance root@YOUR_SERVER_IP:/opt/bupek-finance

# OR use git if hosted on GitHub/GitLab
ssh root@YOUR_SERVER_IP
git clone https://github.com/YOUR_ORG/bupek-finance.git /opt/bupek-finance

cd /opt/bupek-finance/backend

8. Run Database Migrations

cd /opt/bupek-finance/backend
npm ci
npx prisma generate
npx prisma migrate deploy
npx ts-node prisma/seed.ts     # loads initial data + CEO account
✔ After seeding, the CEO login will work: ceo@bupekfinance.co.tz / Admin@1234
Change this password immediately after first login.

9. Build & Start the Application

# Build backend
cd /opt/bupek-finance/backend
npm run build

# Build frontend
cd /opt/bupek-finance/frontend
npm ci
npm run build

Start with PM2

# Start backend
pm2 start /opt/bupek-finance/backend/dist/server.js \
  --name "bupek-api" \
  --cwd /opt/bupek-finance/backend

# Start frontend
pm2 start npm --name "bupek-frontend" \
  --cwd /opt/bupek-finance/frontend \
  -- start

# Save PM2 config (auto-restart on reboot)
pm2 save
pm2 startup   # follow the command it prints

Verify services are running

pm2 status
curl http://localhost:5000/api/v1/health    # should return {"status":"ok"}
curl http://localhost:3000                  # should return HTML

10. SSL Certificate (HTTPS)

sudo apt-get install -y certbot python3-certbot-nginx
sudo certbot --nginx -d yourdomain.co.tz

# Auto-renewal is set up automatically. Test with:
sudo certbot renew --dry-run
✔ After this step, your system will be accessible at https://yourdomain.co.tz with a valid SSL certificate (free, renews automatically every 90 days).

11. Set Up Automatic Backups

# Install cron job for daily 2am backups
bash /opt/bupek-finance/scripts/setup_backup_cron.sh

# Test backup manually
bash /opt/bupek-finance/scripts/backup.sh

# Check backup files
ls -lh /var/backups/bupek/
⚠ Also copy backups to an external location (cloud storage, another server, USB drive) weekly. Local-only backups are lost if the server fails.

12. Monitoring & Maintenance

Common commands

# View running processes
pm2 status

# View backend logs
pm2 logs bupek-api

# View frontend logs
pm2 logs bupek-frontend

# Restart after code update
pm2 restart bupek-api
pm2 restart bupek-frontend

# View nginx access log
sudo tail -100 /var/log/nginx/access.log

# Check disk space
df -h

# View backup log
cat /var/backups/bupek/backup.log

After updating the code

cd /opt/bupek-finance
git pull                           # if using git
cd backend && npm run build && pm2 restart bupek-api
cd ../frontend && npm run build && pm2 restart bupek-frontend
